#include <stdio.h>
#include <math.h>

#define ROWS 20
#define COLS 40

char pic[ROWS][COLS];

void clearPicture()
{
    int i, j;
    for(i = 0; i < ROWS; i++)
    {
        for(j = 0; j < COLS; j++)
        {
            pic[i][j] = '.';
        }
    }
}

void displayPicture()
{
    int i, j;

    for(i = 0; i < ROWS; i++)
    {
        for(j = 0; j < COLS; j++)
        {
            printf("%c ", pic[i][j]);
        }
        printf("\n");
    }
}

int isValid(int x, int y)
{
    return (x >= 0 && x < COLS &&
            y >= 0 && y < ROWS);
}

void drawLine(int x1, int y1, int x2, int y2)
{
    int dx = x2 - x1;
    int dy = y2 - y1;

    int steps = abs(dx) > abs(dy) ? abs(dx) : abs(dy);

    float xInc = dx / (float)steps;
    float yInc = dy / (float)steps;

    float x = x1;
    float y = y1;

    int i;

    for(i = 0; i <= steps; i++)
    {
        if(isValid((int)x, (int)y))
            pic[(int)y][(int)x] = '#';

        x += xInc;
        y += yInc;
    }
}

void drawRectangle(int x1, int y1, int x2, int y2)
{
    int i;

    for(i = x1; i <= x2; i++)
    {
        if(isValid(i, y1)) pic[y1][i] = '#';
        if(isValid(i, y2)) pic[y2][i] = '#';
    }

    for(i = y1; i <= y2; i++)
    {
        if(isValid(x1, i)) pic[i][x1] = '#';
        if(isValid(x2, i)) pic[i][x2] = '#';
    }
}

void drawTriangle(int x, int y, int size)
{
    int i;

    for(i = 0; i < size; i++)
    {
        if(isValid(x-i, y+i))
            pic[y+i][x-i] = '#';

        if(isValid(x+i, y+i))
            pic[y+i][x+i] = '#';
    }

    for(i = x-size+1; i <= x+size-1; i++)
    {
        if(isValid(i, y+size-1))
            pic[y+size-1][i] = '#';
    }
}

void drawCircle(int cx, int cy, int r)
{
    int x, y;

    for(y = -r; y <= r; y++)
    {
        for(x = -r; x <= r; x++)
        {
            int value = x*x + y*y;

            if(value >= r*r-r && value <= r*r+r)
            {
                if(isValid(cx+x, cy+y))
                    pic[cy+y][cx+x] = '#';
            }
        }
    }
}

int main()
{
    clearPicture();

    drawLine(2, 2, 15, 10);
    drawRectangle(20, 2, 35, 8);
    drawTriangle(10, 12, 5);
    drawCircle(30, 14, 4);

    displayPicture();

    return 0;
}

explain me this condition f(value >= r*r-r && value <= r*r+r)
this condition is used in the circle drawing function:
if(value >= r*r-r && value <= r*r+r)
where
value = x*x + y*y;
Why is it used?
The equation of a circle is:
x2+y2=r2x^2+y^2=r^2x2+y2=r2hhhkkkrrr(x)2+(y)2=3.02(x)^2 + (y)^2 = 3.0^2(x)2+(y)2=3.02
So for points exactly on the circle boundary:
x*x + y*y = r*r
But since we're drawing on a character grid, points don't fall exactly on the circle boundary.
Therefore, instead of checking:
value == r*r
we allow a small range:
r*r-r <= value <= r*r+r
Example
If:
r = 4
Then:
r*r = 16
The condition becomes:
12 <= value <= 20
So any point whose x*x + y*y is between 12 and 20 is marked as #.
This creates a smooth circular outline on the grid instead of missing points. 😊

void deleteArea(int x1, int y1, int x2, int y2)
{
    int i, j;

    for(i = y1; i <= y2; i++)
    {
        for(j = x1; j <= x2; j++)
        {
            if(isValid(j, i))
            {
                pic[i][j] = '.';
            }
        }
    }
}

void modifyRectangle()
{
    int oldx1, oldy1, oldx2, oldy2;
    int newx1, newy1, newx2, newy2;

    printf("Enter old rectangle coordinates:\n");
    scanf("%d%d%d%d",
          &oldx1, &oldy1,
          &oldx2, &oldy2);

    deleteArea(oldx1, oldy1, oldx2, oldy2);

    printf("Enter new rectangle coordinates:\n");
    scanf("%d%d%d%d",
